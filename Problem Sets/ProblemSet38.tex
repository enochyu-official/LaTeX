\subsection*{1965 Hungary Math Olympiad Problem 1}
{\color {Gray}
Find integers $a$, $b$, and $c$ such that
\[
a^2 + b^2 + c^2 + 3 < ab + 3b + 2c.
\]
}
\MyLine

\begin{solution}
Completing square method could be used to solve the problem.
\begin{align*}
    \left( a^2 - ab + \frac{1}{4} b^2 \right) + \frac{3}{4} b^2 - 3b + (c - 1)^2 + 2 < 0 \\
    \left( a - \frac{1}{2} b \right)^2 + \frac{3}{4} \left( b^2 - 4b + 4 \right) - 3 + (c - 1)^2 + 2 < 0 \\
    \left( a - \frac{1}{2} b \right)^2 + \frac{3}{4} (b - 2)^2 + (c - 1)^2 < 1
\end{align*}
Therefore, the only possible case is when $\boxed{a = 1, b = 2, c=1}$.
\end{solution}

\subsection*{Problem}
{\color {Gray}
Suppose $x$ and $y$ are real numbers. Find the minimum value of
\[
u = x^2 + 2xy + 5y^2 + 2x - 6y + 7.
\]
}
\MyLine

\begin{solution}
Because there exists $x^2, y^2, xy, x, y$, it is natural to immediately think of the factorization form $(x \pm y + C)^2$.
\begin{align*}
    (x + y + 1)^2 + 4y^2 - 8y + 6 &= 0 \\
    (x + y + 1)^2 + 4(y - 1)^2 + 2 &= 0
\end{align*}
Thus, the minimum value of $u$ is $\boxed{2}$.
\end{solution}

\subsection*{Problem}
{\color {Gray}
Suppose $\frac{1}{2} \leq x^2 + 4y^2 \leq 2$ is true for real numbers $x$ and $y$.
\begin{enumerate}
    \item Find the greatest value of $x^2 - 2xy + 4y^2$
    \item Find the smallest value of $x^2 - 2xy + 4y^2$
\end{enumerate}
}
\MyLine

\subsection*{Part 1.}
\textbf{Solution I.} Completing square method may provide the maximum value.
\begin{align*}
    x^2 + 4y^2
    &= \frac{2}{3} \left( x^2 - 2xy + 4y^2 \right) + \frac{1}{3} \left( x^2 + 4xy + 4y^2 \right) \\
    &= \frac{2}{3} \left( x^2 - 2xy + 4y^2 \right) + \frac{1}{3} (x + 2y)^2
\end{align*}
Continuing,
\[
\frac{1}{2} \leq \frac{2}{3} \left( x^2 - 2xy + 4y^2 \right) + \frac{1}{3} (x + 2y)^2 \leq 2
\]
Notice that when $x = -2y$, $x^2 - 2xy + 4y^2 \leq 3$. However, other cases will lead to $x^2 - 2xy + 4y^2$ being less than or equal to a value smaller than $3$. Therefore, $x^2 - 2xy + 4y^2$ will reach the maximum $3$ when $x = -2y$, which is the case that provides the maximum value.
\qed

\textbf{Solution II.} Using AM-GM inequality, it is evident that $x^2 + 4y^2 \geq |4xy|$. In other words, $|2xy| \leq 1$. Moreover,
\[
\frac{1}{2} - 2xy \leq x^2 - 2xy + 4y^2 \leq 2 - 2xy
\]
Consider the case when $x = -1$ and $y = \frac{1}{2}$. $x^2 + 4y^2$ will reach its maximum while $xy$ will face its minimum. Thus, $2 + 1$, or $\boxed{3}$ is the maximum value.
\qed

\subsection*{Part 2.}
\begin{solution}
Completing square method could be used.
\begin{align*}
    x^2 + 4y^2
    &= 2 \left( x^2 - 2xy + 4y^2 \right) - (x^2 - 4xy + 4y^2) \\
    &= 2 \left( x^2 - 2xy + 4y^2 \right) - (x - 2y)^2
\end{align*}
Therefore,
\[
\frac{1}{2} \leq 2 \left( x^2 - 2xy + 4y^2 \right) - (x - 2y)^2 \leq 2
\]
Notice that $x^2 - 2xy + 4y^2$ will reach its minimum \textit{iff} $x = 2y$. Thus, the minimum value of $x^2 - 2xy + 4y^2$ is $\boxed{\frac{1}{4}}$.
\end{solution}
